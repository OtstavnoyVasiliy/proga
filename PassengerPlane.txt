public class PassengerPlane extends Airplane{
    int numOfSeats;

    public PassengerPlane() {
        super();
        numOfSeats = 200;
    }

    public PassengerPlane(String brand, double fuelTankVolume, int rangeOfFlight, int numOfSeats) {
        super(brand, fuelTankVolume, rangeOfFlight);
        this.numOfSeats = numOfSeats;
    }

    void print() {
        super.print();
        System.out.println(", numOfSeats="+numOfSeats);
    }

    public int getNumOfSeats() {
        return numOfSeats;
    }

    public void setNumOfSeats(int numOfSeats) {
        this.numOfSeats = numOfSeats;
    }
}
