public class Man {
    private String fio;
    private int year;
    Man(){
        fio="";
        year=-1;
    }
    Man(String fio,int year){
        this.fio=fio;
        this.year=year;
    }

    public String getFio() {
        return fio;
    }

    @Override
    public String toString() {
        return "Man{" +
                "fio='" + fio + '\'' +
                ", year=" + year +
                '}';
    }

    public void setFio(String fio) {
        this.fio = fio;
    }

    public int getYear() {
        return year;
    }

    public void setYear(int year) {
        this.year = year;
    }

    int age(int current_year){
        return current_year-year;
    }
}
